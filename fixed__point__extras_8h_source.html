57  template<class Rep>
58  constexpr Rep sqrt_solve3(
59  Rep n,
60  Rep bit,
61  Rep result)
62  {
63  return (bit!=Rep{0})
64  ? (n>=result+bit)
65  ? sqrt_solve3<Rep>(
66  static_cast<Rep>(n-(result+bit)),
67  bit >> 2,
68  static_cast<Rep>((result >> 1)+bit))
69  : sqrt_solve3<Rep>(n, bit >> 2, result >> 1)
70  : result;
71  }
