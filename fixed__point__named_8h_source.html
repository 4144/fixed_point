117  // 'Wide' doesn't guarantee avoiding overflow. Adding a single bit to add/subtract results would often lead to double the width being used.
118  static constexpr int sufficient_sign_bits = std::is_signed<rep_op_result>::value;
119  static constexpr int sufficient_integer_digits = _impl::max(Lhs::integer_digits,

121  static constexpr int sufficient_fractional_digits = _impl::max(Lhs::fractional_digits,
122  Rhs::fractional_digits);
123  static constexpr _width_type sufficient_width =
124  sufficient_sign_bits+sufficient_integer_digits+sufficient_fractional_digits;
125  static constexpr int result_width = _impl::max(sufficient_width, width<rep_op_result>::value);
126 
127  using rep_type = set_width_t<rep_op_result, result_width>;
128  using type = fixed_point<rep_type, -sufficient_fractional_digits>;
